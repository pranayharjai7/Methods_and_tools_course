5a. Decision boundary — linear case¶

For logistic regression with 2 normalized features. Decision boundary is where w0 + w1*x1_norm + w2*x2_norm = 0 → rearrange for x2.

In [15]:
# dummy weights and normalization stats
w    = np.array([[0.5], [2.0], [1.5]])   # [w0, w1, w2]
mean = np.array([0.0, 0.0])
std  = np.array([1.0, 1.0])
X_original = X_cls

# x-range in original scale
x1_vals = np.array([X_original[:, 0].min() - 1, X_original[:, 0].max() + 1])

# normalize x1 range
x1_norm = (x1_vals - mean[0]) / std[0]

# solve for x2_norm: w0 + w1*x1_norm + w2*x2_norm = 0
x2_norm = (-w[0] - w[1]*x1_norm) / w[2]

# convert x2 back to original scale
x2_vals = x2_norm * std[1] + mean[1]

plt.scatter(pos[:, 0], pos[:, 1], c='g', marker='o', label='Class 1')
plt.scatter(neg[:, 0], neg[:, 1], c='r', marker='x', label='Class 0')
plt.plot(x1_vals, x2_vals, 'k-', label='Decision boundary')
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('Decision boundary (linear)')
plt.legend()
plt.show()

5b. Decision boundary — non-linear case (contour)¶

When features are polynomial-expanded (Lab04 style). Plot the contour where the model output = 0 (decision boundary).

In [20]:
def mapFeature(X1, X2, deg):
    # Build polynomial feature map: [1, x1, x2, x1^2, x1*x2, x2^2, ...]
    try:
        m = X1.shape[0]
    except:
        m = 1
        X1 = np.array([X1])
        X2 = np.array([X2])
    out = np.ones((m, 1))
    for i in range(1, deg + 1):
        for j in range(i + 1):
            term = ((X1**(i-j)) * (X2**j)).reshape(m, 1)
            out = np.hstack((out, term))
    return out

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# dummy weights for degree=3 expanded features (10 features total)
w_nonlin = np.random.randn(10, 1) * 0.5

u_vals = np.linspace(-2, 2, 60)
v_vals = np.linspace(-2, 2, 60)
z = np.zeros((len(u_vals), len(v_vals)))

for i in range(len(u_vals)):
    for j in range(len(v_vals)):
        z[i, j] = (mapFeature(u_vals[i], v_vals[j], 3) @ w_nonlin)[0, 0]

plt.scatter(pos[:, 0], pos[:, 1], c='g', marker='o', label='Class 1')
plt.scatter(neg[:, 0], neg[:, 1], c='r', marker='x', label='Class 0')
plt.contour(u_vals, v_vals, z.T, levels=[0], colors='black')   # contour at z=0
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('Decision boundary (non-linear, contour at z=0)')
plt.legend()
plt.show()

8. Histogram vs theoretical PDF — Problem 2 style¶

Show sampled distribution matches the theoretical one.

In [23]:
# Example: Laplace distribution (from Problem 2)
lam = 3
mu  = 0
N   = 50000

rng = np.random.default_rng(seed=42)
u1  = rng.uniform(0, 1, N)
u2  = rng.uniform(0, 1, N)
s   = np.where(u1 < 0.5, 1, -1)
x_samples = mu + s * lam * np.log(u2) / np.sqrt(2)

# theoretical PDF
x_plot = np.linspace(-15, 15, 500)
A      = 1 / (lam * np.sqrt(2))
pdf    = A * np.exp(-np.sqrt(2) * np.abs(x_plot - mu) / lam)

plt.figure(figsize=(8, 5))
plt.hist(x_samples, bins=80, density=True,              # density=True normalizes to area=1
         alpha=0.5, color='steelblue', label='Sampled histogram')
plt.plot(x_plot, pdf, 'r-', linewidth=2.5, label='Theoretical PDF')
plt.xlabel('x')
plt.ylabel('Probability density')
plt.title(f'Laplace distribution: λ={lam}, μ={mu}')
plt.legend()
plt.xlim(-15, 15)
plt.grid(True, alpha=0.3)
plt.show()

Matplotlib quick reference¶

plt.figure(figsize=(8, 5))          # set figure size
plt.scatter(x, y)                   # scatter plot
plt.plot(x, y, 'r-', linewidth=2)   # line plot
plt.fill_between(x, y1, y2, alpha=0.2)  # shaded band
plt.contour(u, v, z, levels=[0])    # contour line (decision boundary)
plt.hist(x, bins=80, density=True)  # histogram normalized to area=1
plt.xlabel('label')
plt.ylabel('label')
plt.title('title')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()                          # always end with this

Common colors: 'steelblue', 'r' (red), 'g' (green), 'k' (black)
Common markers: 'o' (circle), 'x' (cross), 's' (square)
zorder: higher = drawn on top (use zorder=5 for data dots, zorder=4 for lines)
